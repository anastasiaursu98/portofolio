In your layout, the flex container wraps both the avatar and the text, so the text pushing wider can indirectly affect the avatar.

Here’s a clean way to fix it:

1. Force the avatar container to never shrink:

tsx
<div className="flex-shrink-0 w-10 h-10 rounded-full overflow-hidden">
  <Image src={Avatar} alt="User" className="w-10 h-10 rounded-full object-cover" />
</div>

  • flex-shrink-0 ensures the avatar doesn’t shrink.

  • overflow-hidden + object-cover ensures the image keeps its aspect ratio and fills the container without stretching.

2. Limit the width of the text container

Add a max-w to the text container to prevent it from forcing the parent flex container wider than the sidebar:

tsx
<div className="flex flex-col gap-1 max-w-[calc(100%-3rem)]">
  ...
</div>

  • Here 3rem roughly equals the avatar width + gap (w-10 + gap-3), so text doesn’t push the avatar.

3. Ensure text truncation works properly

You already have truncateMiddle and truncateNameFirstFullLastMiddle. Add overflow-hidden and whitespace-nowrap to the text divs to prevent them from wrapping and stretching the container:

tsx
<div

>
  {truncateNameFirstFullLastMiddle(userData?.name || "User", 25)}

  title={userData?.email || "user@example.com"}

</div>

"use client";

import Image from "next/image";
import { useRouter } from "next/navigation";
import { useEffect, useState } from "react";

import { userStorage } from "@/features/auth/services/userStorage";
import { useAuthStore } from "@/features/auth/store/auth";
import {
  PRIVATE_ROUTES,
  PUBLIC_ROUTES,
} from "@/features/auth/constants/routes";
import TooltipCustome from "@/components/shared/TooltipCustom";
import Avatar from "@/assets/images/avatar.png";
import { LogOutIcon } from "@/features/auth/assets/LogOutIcon";

export const AppSidebarFooter = () => {
  const router = useRouter();
  const { logout } = useAuthStore();
  const [userData, setUserData] = useState<{ name: string; email: string } | null>(null);

  useEffect(() => {
    const storedUserData = userStorage.get();
    if (storedUserData && storedUserData.name && storedUserData.email) {
      setUserData({
        name: storedUserData.name,
        email: storedUserData.email,
      });
    }
  }, []);

  const handleLogout = async () => {
    await fetch(PRIVATE_ROUTES.LOGOUT, { method: "POST" });
    userStorage.clear();
    logout();
    router.push(PUBLIC_ROUTES.LOG_IN);
  };

  function truncateMiddle(text: string, maxLength: number): string {
    if (text.length <= maxLength) return text;

    const separator = "...";
    const charsToShow = maxLength - separator.length;
    const frontChars = Math.ceil(charsToShow / 2);
    const backChars = Math.floor(charsToShow / 2);

    return text.slice(0, frontChars) + separator + text.slice(text.length - backChars);
  }

  function truncateNameFirstFullLastMiddle(fullName: string, maxLength: number): string {
    if (fullName.length <= maxLength) return fullName;

    const parts = fullName.trim().split(/\s+/);
    if (parts.length >= 2) {
      const first = parts[0];
      const last = parts[parts.length - 1];
      const remainingLength = maxLength - (first.length + 1);
      const shortenedLast = truncateMiddle(last, remainingLength);
      return `${first} ${shortenedLast}`;
    }
    return truncateMiddle(fullName, maxLength);
  }

  return (
    <div className="bg-white rounded-xl shadow flex items-center p-3 gap-3">
      {/* Avatar */}
      <div className="flex-shrink-0 w-10 h-10 rounded-full overflow-hidden">
        <Image src={Avatar} alt="User" className="w-10 h-10 rounded-full object-cover" />
      </div>

      {/* User Info */}
      <div className="flex flex-col gap-1 max-w-[calc(100%-3rem)]">
        <div className="flex items-center justify-between gap-2">
          <TooltipCustome content={userData?.name || "User"} trigger="hover" position="top">
            <div
              className="font-semibold text-[15px] cursor-pointer overflow-hidden whitespace-nowrap"
              title={userData?.name || "User"}
            >
              {truncateNameFirstFullLastMiddle(userData?.name || "User", 25)}
            </div>
          </TooltipCustome>

          <span
            className="cursor-pointer inline-block transition-transform duration-200 ease-in-out hover:scale-110 active:scale-95"
            onClick={handleLogout}
          >
            <LogOutIcon />
          </span>
        </div>

        <TooltipCustome content={userData?.email || "user@example.com"} trigger="hover" position="top">
          <div
            className="text-xs text-gray-500 cursor-pointer overflow-hidden whitespace-nowrap"
            title={userData?.email || "user@example.com"}
          >
            {truncateMiddle(userData?.email || "user@example.com", 25)}
          </div>
        </TooltipCustome>
      </div>
    </div>
  );
};

✅ What changed:

  1. Avatar container: flex-shrink-0 w-10 h-10 overflow-hidden prevents stretching.

  2. Image: object-cover ensures the avatar maintains its aspect ratio.

  3. Text container: max-w-[calc(100%-3rem)] prevents text from pushing the avatar.

  4. Text elements: overflow-hidden whitespace-nowrap prevents long names/emails from wrapping or expanding the container.
